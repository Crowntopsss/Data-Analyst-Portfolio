join Genre
on track.GenreId = Genre.GenreId 
join InvoiceLine
on track.TrackId = InvoiceLine.TrackId
join Invoice
on InvoiceLine.InvoiceId = Invoice.InvoiceId
join customer
on Invoice.CustomerId = Customer.CustomerId
